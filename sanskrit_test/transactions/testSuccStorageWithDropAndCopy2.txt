signed(A) transaction{
   a = lit 1:IntU8.U8;
   b = pack SomeTypes.X[IntU8.U8]#Ctr(a);
   d = call IndexOps.ModIndex();
   (r, s) = call SomeTypes.GenIndexedAt[SomeTypes.X[IntU8.U8]](d,b);
   store s;
};

signed(A) transaction{
   orig = lit 1:IntU8.U8;
   a = call IndexOps.ModIndex();
   r = call SomeFunctions.GenRef(a);
   b = &load r;
   (d0, i0) = call SomeTypes.BorrowEntryX[IntU8.U8](b);
   d2 = &unpack i0:SomeTypes.X#Ctr;
   () = call Helpers.AssertEqU8(d2,orig);
};
